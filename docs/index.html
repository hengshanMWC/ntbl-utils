utils

API Documentation

Installation

npm i @ntbl/utils --save

Usage

// 导入所有的工具函数
import utils from '@ntbl/utils'

// 导入模块的所有工具函数
import functions from '@ntbl/utils/functions'

// 导入一个工具函数
import agent from '@ntbl/utils/agent'
